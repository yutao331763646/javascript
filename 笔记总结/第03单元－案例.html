function hash(num){
        var n1 = (num%10+5)%10;
        var n2 = (parseInt(num%100/10)+5)%10;
        var n3 = (parseInt(num%1000/100)+5)%10;
        var n4 = (parseInt(num/1000)+5)%10;
        return n1*1000+n2*100+n3*10+n4;
}
