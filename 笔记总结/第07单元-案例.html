动态表格创建

onload = function(){
    var btn = document.getElementById("btn");
    btn.onclick = function(){
        var tab = document.createElement("table");
        var rows = document.getElementById("row").value;
        var cols = document.getElementById("col").value;
        for(var i=0; i<rows; i++) { //行数
            var tr = document.createElement("tr");//创建一行
            (function(){
                for(var i=0; i<cols; i++) { //列数
                    var td = document.createElement("td");//创建一列
                    td.innerHTML = Math.round( Math.random()*90+10 ); //单元格填写随机数字
                    td.style.background = "#"+randomColor();
                    tr.appendChild(td);//将td放入tr
                }
            })();
            tab.appendChild(tr);
        }
        document.body.appendChild(tab);
    }
}
